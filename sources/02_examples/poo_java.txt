                    01   package paquete;
                    02   import java.util.Scanner;
                    03 
                    04   public class Clase {
                    05 
                    06       public static void main(String[] args) {
                    07           // Objetivo del Algoritmo: Analizar el Objetivo / Problema como un Sistema [Entradas, Proceso y Salidas]
                    08           System.out.println("Ingresar Datos:\n");
                    09           
                    10           // Declarar variables, constantes, arreglos u objetos.
                    11           int numero;
                    12           String nombre;
                    13           char letra;
                    14           double temperatura;
                    15           boolean estado;
                    16           long numeroLargo;
                    17           Scanner in = new Scanner(System.in);
                    18           
                    19           // Dimensionar: arreglos.
                    20           
                    21           // Iniciar: variables, constantes y/o arreglos        
                    22           
                    23           // Entrada de Datos: Pueden ser solicitados o no        
                    24           System.out.print("Ingrese un número: ");
                    25           numero = in.nextInt();
                    26           System.out.print("Ingrese un nombre: ");
                    27           nombre = in.next();
                    28           System.out.print("Ingrese una letra: ");
                    29           letra = in.next().charAt(0);
                    30           System.out.print("Ingrese un valor de temperatura: ");
                    31           temperatura = in.nextDouble();
                    32           System.out.print("Ingrese un estado (Verdadero = true, Falso = false): ");
                    33           estado = in.nextBoolean();
                    34           System.out.print("Ingrese un número Largo: ");
                    35           numeroLargo = in.nextLong();
                    36           
                    37           // Proceso: Generalmente, se usan las estructuras de control: Secuencial, condicional, repetición.
                    38           
                    39           // Salida de Datos: Resultado(s).
                    40           System.out.println("\nEl número ingresado fue: " + numero);        
                    41           System.out.println("El nombre ingresado fue: " + nombre);        
                    42           System.out.println("La letra ingresada fue: " + letra);
                    43           System.out.println("La temperatura ingresada fue: " + temperatura);
                    44           System.out.println("El estado ingresado fue: " + estado);
                    45           System.out.println("El número largo ingresado fue: " + numeroLargo);
                    46       }
                    47       
                    48   }
